function fruit(title,weight){
this.title=title;
this.weight=weight;
}



var apple = new fruit("apple",200);

var banana = new fruit("banana",300);


var fruitArray = [];
fruitArray.push(apple);
fruitArray.push(banana);


for( var i = 0; i< fruitArray.length; i++){
document.getElementById("demo").innerHTML += "
" +
fruitArray[i].title+ " has weight " + fruitArray[i].weight;
}
